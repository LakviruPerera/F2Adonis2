#' Permutational Multivariate Analysis of Variance Using the F2 Statistic
#'
#' Analysis of variance using distance matrices for partitioning variation
#' among sources. Unlike standard PERMANOVA, this function uses the F2 statistic
#' (Anderson 2017), which accounts for heterogeneous dispersions (Behrens-Fisher problem).
#'
#' @param formula Model formula. The LHS must be a community data matrix or a
#'    dissimilarity matrix. The RHS must define a single factor predictor.
#' @param data A data frame containing the independent variables.
#' @param permutations Number of permutations for the test, or a permutation matrix.
#' @param method Pairwise distance method used in \code{vegdist} (default "euclidean").
#' @param sqrt.dist Logical; take square root of dissimilarities to "euclidify" them.
#' @param add Add a constant to dissimilarities to avoid negative eigenvalues.
#'    Options are "lingoes" or "cailliez".
#' @param by Validation restricted to \code{NULL}. F2 currently only supports global tests.
#' @param parallel Number of parallel processes to use.
#' @param na.action Handling of missing values on the RHS.
#' @param strata Groups within which to constrain permutations.
#' @param bootstrap Logical; perform bias-adjusted bootstrapping (Anderson 2017).
#' @param bias.adjust Logical; apply bias correction to bootstrap variances.
#' @param ... Other arguments passed to \code{vegdist}.
#'
#' @details
#' \code{F2adonis2} implements the multivariate F2 statistic and the
#' bias-adjusted bootstrap procedure as derived in Anderson et al. (2017).
#' This approach addresses the multivariate Behrens-Fisher problem, where
#' groups have unequal dispersions (heteroscedasticity) and unbalanced
#' sample sizes.
#'
#' Standard PERMANOVA (the F1 statistic) can lead to inflated Type I error
#' rates if the smaller group has a larger dispersion. The F2 statistic
#' corrects this by using group-specific mean-squared dissimilarities (Vi)
#' in the denominator.
#'
#' @section Bootstrapping:
#' If \code{bootstrap = TRUE}, the function performs resampling of residuals
#' from a common centroid. If \code{bias.adjust = TRUE}, it corrects for the
#' inherent overestimation of variance in bootstrap samples by comparing
#' bootstrap variances against observed group variances.
#'
#' @return Returns an \code{anova.cca} result object with an additional column
#'    for \code{F2}. If bootstrapping is enabled, columns for bootstrap p-values
#'   are included.
#'
#' @references
#' Anderson, M. J., Walsh, D. C., Clarke, K. R., Gorley, R. N., & Guerra-Castro, E. (2017).
#' Some solutions to the multivariate Behrens–Fisher problem for dissimilarity-based
#' analyses. \emph{Australian & New Zealand Journal of Statistics}, 59(1), 57-79.
#' \doi{10.1111/anzs.12176}
#'
#' @author Lakviru Perera (based on logic by Marti J. Anderson)
#'
#' @seealso \code{\link[vegan]{adonis2}}, \code{\link[vegan]{betadisper}}
#'
#' @examples
#' library(vegan)
#' data(dune)
#' data(dune.env)
#' # F2 test with Bias-Adjusted Bootstrap
#' F2adonis2(dune ~ Management, data = dune.env, bootstrap = TRUE, bias.adjust = TRUE)
#'
#' @export
F2adonis2 <-
  function(formula, data, permutations = 999, method = "euclidean",
           sqrt.dist = FALSE, add = FALSE, by = NULL,
           parallel = getOption("mc.cores"), na.action = na.fail,
           strata = NULL, bootstrap = FALSE, bias.adjust = FALSE, ...)
  {
    # 1. SCOPING: Capture Environment correctly
    if (missing(data))
      data <- parent.frame()
    else
      data <- eval(match.call()$data, parent.frame(),
                   enclos = environment(formula))

    # 2. VALIDATION: 'by' operator restriction (Anderson 2017 global test focus)
    if (!is.null(by)) {
      stop("\nError: F2 PERMANOVA currently only supports by = NULL (Global test).\n",
           "Sequential or Marginal terms for F2 require specific denominator\n",
           "adjustments not yet implemented.", call. = FALSE)
    }

    # 3. PARSE Formula for counting
    formula_obj <- formula(terms(formula, data = data))
    term_labels <- attr(terms(formula_obj), "term.labels")
    n_predictors <- length(term_labels)

    # 4. VALIDATION: Predictor Count Check (Using labels, not matrix columns)
    if (n_predictors > 1) {
      stop("F2adonis2 currently supports only ONE predictor variable.\n",
           "You have ", length(term_labels), " predictors in your model.\n\n",
           "Combine multiple variables into a single predictor and run the function again.", call. = FALSE)
    }

    # 5. VALIDATION: Factor Check
    predictor_name <- term_labels[1]
    # Handle factor transformation inside formula
    predictor_val <- tryCatch({
      eval(parse(text = predictor_name), envir = data)
    }, error = function(e) data[[predictor_name]])

    if (!is.factor(predictor_val)) {
      stop("The predictor variable must be a factor.\n",
           "Create a factor version of your variable (e.g. data$var <- as.factor(data$var))\n",
           "or refit the model using the factor() function inside the formula.", call. = FALSE)
    }

    # Standard Setup continues
    lhs <- eval(formula_obj[[2]], envir = parent.frame(), enclos = environment(formula_obj))

    original_data <- NULL
    if (!inherits(lhs, "dist") && (is.matrix(lhs) || is.data.frame(lhs)) && !isSymmetric(unname(as.matrix(lhs)))) {
      original_data <- as.matrix(lhs)
    }
    original_method <- method

    if ((is.matrix(lhs) || is.data.frame(lhs)) && isSymmetric(unname(as.matrix(lhs))))
      lhs <- as.dist(lhs)
    if (!inherits(lhs, "dist"))
      lhs <- vegdist(as.matrix(lhs), method=method, ...)

    d <- vegan:::ordiParseFormula(formula = formula_obj, data = data, na.action = na.action, subset = NULL, X = lhs)

    if (is.null(d$Y)) stop("needs explanatory variables on the right-hand-side")

    lhs_val <- d$X
    if (!is.null(d$na.action)) {
      lhs_val <- lhs_val[-d$na.action, -d$na.action, drop = FALSE]
      if (!is.null(original_data)) original_data <- original_data[-d$na.action, , drop = FALSE]
    }

    if (sqrt.dist) lhs_val <- sqrt(lhs_val)
    if (is.logical(add) && add) add <- "lingoes"
    if (is.character(add)) {
      add <- match.arg(add, c("lingoes", "cailliez"))
      if (add == "lingoes") {
        ac <- vegan:::addLingoes(as.matrix(lhs_val))
        lhs_val <- sqrt(lhs_val^2 + 2 * ac)
      } else if (add == "cailliez") {
        ac <- vegan:::addCailliez(as.matrix(lhs_val))
        lhs_val <- lhs_val + ac
      }
    }

    sol <- F2adonis0(lhs_val, d$Y, d$Z, original_data, original_method)
    sol$formula <- match.call()
    sol$terms <- d$terms
    sol$terminfo <- vegan:::ordiTerminfo(d, data)

    perm <- vegan:::getPermuteMatrix(permutations, NROW(lhs_val), strata = strata)
    out <- anova(sol, permutations = perm, by = by, parallel = parallel, bootstrap = bootstrap, bias.adjust = bias.adjust)

    att <- attributes(out)
    att$heading[1] <- "Permutation test for F2-based PERMANOVA"
    att$heading[2] <- deparse(match.call(), width.cutoff = 500L)
    attributes(out) <- att
    out
  }

# ==============================================================================
# F2adonis0 & ANOVA METHODS (Anderson 2017 Logic)
# ==============================================================================

`F2adonis0` <-
  function(lhs, X, Z, original_data = NULL, original_method = "euclidean")
  {
    G <- vegan:::initDBRDA(lhs)
    sol <- list(Ybar = G, tot.chi = sum(diag(G)), adjust = 1, method = "F2_adonis")
    sol$dmat_squared <- as.matrix(lhs)^2
    sol$original_data <- original_data
    sol$original_method <- original_method
    sol$X_original <- X

    if (!is.null(Z) && ncol(Z)) {
      Z <- scale(Z, scale = FALSE)
      QZ <- qr(Z); G <- qr.resid(QZ, G); G <- qr.resid(QZ, t(G))
      pCCA <- list(rank = QZ$rank, tot.chi = rowSums(qr.fitted(QZ, G)), QR = QZ)
    } else { pCCA <- NULL }

    if (!is.null(Z)) X <- cbind(Z, X)
    X <- scale(X, scale = FALSE)
    qrhs <- qr(X)
    Gfit <- qr.fitted(qrhs, G); Gfit <- qr.fitted(qrhs, t(Gfit))
    Gres <- qr.resid(qrhs, G); Gres <- qr.resid(qrhs, t(Gres))

    rank <- qrhs$rank
    if (!is.null(Z)) rank <- rank - QZ$rank
    if(rank > 0) CCA <- list(rank = rank, qrank = rank, tot.chi = sum(diag(Gfit)), QR = qrhs)
    else CCA <- NULL

    CA <- list(rank = nrow(lhs) - max(qrhs$rank, 0) - 1, u = matrix(0, nrow = nrow(lhs)), tot.chi = sum(diag(Gres)))
    if (!is.null(pCCA)) sol$pCCA <- pCCA
    sol$CCA <- CCA; sol$CA <- CA
    class(sol) <- c("F2adonis2", "dbrda", "rda", "cca")
    sol
  }

#' @export
anova.F2adonis2 <- function(object, permutations, by = NULL,
                            parallel = 1, bootstrap = FALSE,
                            bias.adjust = FALSE, ...) {
  X_orig <- object$X_original
  groups <- if (ncol(X_orig) == 1) as.factor(X_orig[, 1]) else interaction(as.data.frame(X_orig), drop = TRUE)
  G_obs <- object$Ybar; dmat_squared <- object$dmat_squared; n <- nrow(dmat_squared)
  H_obs <- get_H(model.matrix(~ groups)); SS_A_obs <- sum(diag(H_obs %*% G_obs))
  res_obs <- calculate_F2_and_Vi(dmat_squared, groups, SS_A_obs, n)
  F2_obs <- res_obs$F2; Vi_obs <- res_obs$Vi

  n_perms <- nrow(permutations); F2_perm <- numeric(n_perms)
  for (i in 1:n_perms) {
    grp_perm <- groups[permutations[i, ]]
    H_p <- get_H(model.matrix(~ grp_perm)); SS_A_p <- sum(diag(H_p %*% G_obs))
    F2_perm[i] <- calculate_F2_and_Vi(dmat_squared, grp_perm, SS_A_p, n)$F2
  }
  p_perm <- (sum(F2_perm >= F2_obs, na.rm = TRUE) + 1) / (n_perms + 1)

  p_boot <- NA; p_ba_boot <- NA
  if (bootstrap && !is.null(object$original_data)) {
    Y_data <- object$original_data; method <- object$original_method
    F2_boot <- numeric(n_perms); Vi_boot_matrix <- matrix(NA, n_perms, length(unique(groups)))
    SS_A_boot_dist <- numeric(n_perms) # Storage for loop-specific SS (Document 8)

    centroid_common <- colMeans(Y_data); R_obs <- sweep(Y_data, 2, centroid_common, "-")
    for (i in 1:n_perms) {
      Y_b <- NULL; grp_b <- NULL
      for (g in levels(groups)) {
        idx_g <- which(groups == g); R_obs_g <- R_obs[idx_g, , drop = FALSE]
        boot_idx <- sample(nrow(R_obs_g), replace = TRUE)
        Y_b <- rbind(Y_b, sweep(R_obs_g[boot_idx, , drop = FALSE], 2, centroid_common, "+"))
        grp_b <- c(grp_b, rep(g, length(idx_g)))
      }
      grp_b <- factor(grp_b, levels = levels(groups))
      d_b <- if (method == "bray") as.matrix(vegdist(Y_b, method = "bray"))^2 else as.matrix(dist(Y_b, method = method))^2
      ss_a_b <- sum(diag(get_H(model.matrix(~ grp_b)) %*% compute_G(d_b)))
      res_b <- calculate_F2_and_Vi(d_b, grp_b, ss_a_b, n)
      F2_boot[i] <- res_b$F2; Vi_boot_matrix[i, ] <- res_b$Vi
      SS_A_boot_dist[i] <- ss_a_b # Dynamic SS capture
    }
    p_boot <- (sum(F2_boot >= F2_obs, na.rm = TRUE) + 1) / (n_perms + 1)
    if (bias.adjust) {
      bias <- colMeans(Vi_boot_matrix, na.rm = TRUE) - Vi_obs
      f2_ba <- sapply(1:n_perms, function(i) {
        denom <- sum((1 - (as.numeric(table(groups))/n)) * (Vi_boot_matrix[i,] - bias))
        # Bootstrap SS_A used for bias adjustment
        if(denom > 0) SS_A_boot_dist[i] / denom else NA
      })
      p_ba_boot <- (sum(f2_ba >= F2_obs, na.rm = TRUE) + 1) / (n_perms + 1)
    }
  }

  out <- data.frame(Df = object$CCA$rank, SumOfSqs = object$CCA$tot.chi, F2 = F2_obs, "Pr(>F2)" = p_perm, stringsAsFactors = FALSE, check.names = FALSE)
  if (!is.na(p_boot)) out$`Pr(>F2.boot)` <- p_boot
  if (!is.na(p_ba_boot)) out$`Pr(>F2.ba)` <- p_ba_boot
  n_cols <- ncol(out); resid_row <- c(object$CA$rank, object$CA$tot.chi, rep(NA, n_cols - 2)); total_row <- c(n - 1, object$tot.chi, rep(NA, n_cols - 2))
  out <- rbind(Model = out, Residual = resid_row, Total = total_row)
  out <- cbind(out[, 1:2], "R2" = out[, 2] / object$tot.chi, out[, 3:ncol(out)])
  class(out) <- c("anova.cca", "anova", "data.frame")
  return(out)
}

compute_G <- function(dmat) {
  n <- nrow(dmat)
  A <- -0.5 * dmat
  I <- diag(n); J <- matrix(1, n, n); C <- I - (1/n) * J
  return(C %*% A %*% C)
}

get_H <- function(rhs) {
  qrhs <- qr(rhs)
  X_full_rank <- rhs[, 1:qrhs$rank, drop = FALSE]
  Q <- qr.Q(qr(X_full_rank))
  return(tcrossprod(Q[, 1:ncol(Q)]))
}

calculate_F2_and_Vi <- function(dmat, grp_ID, SS.A, n) {
  n_per_group <- table(grp_ID)
  VL <- rep(0, length = length(unique(grp_ID)))
  names(VL) <- names(n_per_group)
  for (i in 1:(n - 1)) {
    for (j in (i + 1):n) {
      if (grp_ID[i] == grp_ID[j]) {
        g <- as.character(grp_ID[i])
        nL <- n_per_group[g]
        if (nL > 1) VL[g] <- VL[g] + (dmat[i, j] / (nL * (nL - 1)))
      }
    }
  }
  denom <- sum((1 - (as.numeric(n_per_group) / n)) * VL)
  F2_stat <- if (denom > 0) SS.A / denom else NA
  return(list(F2 = F2_stat, Vi = VL))
}
